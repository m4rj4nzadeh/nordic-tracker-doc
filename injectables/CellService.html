import { HttpService, Injectable } from '@nestjs/common';
import { InjectModel, Model } from 'nestjs-dynamoose';
import { Cell, CellKey } from './cell.interface';

@Injectable()
export class CellService {
  constructor(
    @InjectModel('cell')
    private readonly cellModel: Model<Cell, CellKey>,
    private httpService: HttpService
  ) {}

  create(cellId: string) {
    // @TODO: TEST Create Service Flow

    const splitted = cellId.split("-");
    const cid = Number(splitted[0]);
    const mcc = Number(splitted[1]);
    const mnc = Number(splitted[2]);
    const lac = Number(splitted[3]);
    const current_datetime = new Date();
    const formatted_date = current_datetime.getFullYear() + '-' + (current_datetime.getMonth() + 1) + '-' + current_datetime.getDate() + ' ' + current_datetime.getHours() + ':' + current_datetime.getMinutes() + ':' + current_datetime.getSeconds();
   
    const  cell = this.httpService.post('https://us1.unwiredlabs.com/v2/process.php', {
      "token": process.env.UNWIRED_API,
      "mcc": mcc,
      "mnc": mnc,
      "cells": [{
        "lac": lac,
        "cid": cid
      }],
      "expireDate": formatted_date
    }, undefined);
    return cell;
  }


  findByCellId(cellId: string) {

     return this.cellModel.scan("cellId").eq(cellId).exec();

  }

}
